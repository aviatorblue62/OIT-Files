Task #2

clear all
format short

XX = zeros(1,512);
REF = zeros(1,257);
IMF = zeros(1,257);

n = input('Number of Frequencies');
for k = 1:n
    XX(k) = input('Input Frequencies: ');
end

% Mythical Program which loads data into REF and IMF

N = length(XX);

for k = 1:257
    REF(k) = REF(k)/(N/2);
    IMF(k) = IMF(k)/(N/2);
end

for h = 1:512
    XX(h) = 0;
end
for k = 1:257
    for h = 1:512
        XX(h) = XX(h) + REF(k)*cos(2*pi*k*h/N);
        XX(h) = XX(h) + IMF(k)*sin(2*pi*k*h/N);
    end
end
